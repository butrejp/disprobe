# Example distros.txt demonstrating all features for disprobe (formerly ventoychecker)
# Blank lines and lines starting with # are ignored.

# Distro=LocalVersion

# Simple entries (default source: distrowatch)
fedora=43
ubuntu=22.04
tinycore=15.0

# Distrowatch lookup by distro name (default)
alma=9.1

# Intentionally malformed entry to exercise error handling (no fix required)
almalinux=8.5

# RSS feed override with optional regex (first capture group preferred)
# - source=rss forces RSS fetch
# - feed= explicit RSS URL
# - regex= Python regex to extract version from item title or content
bazzite=43;source=rss;feed=https://distrowatch.com/news/distro/bazzite.xml;regex=(\d+)

# RSS feed without regex (parser will try item titles / distribution release patterns)
someproject=1.0;source=rss;feed=https://example.org/news/someproject.xml

# URL source with regex: scrape arbitrary page and extract version via regex
customos=2.0;source=url;url=https://customos.example/releases;regex=Latest release:\s*v?([0-9]+\.[0-9]+(?:\.[0-9]+)?)

# URL source without regex: fallback to generic scraping heuristics
weirdos=0.9;source=url;url=https://weirdos.example/releases

# Example of a line that will be ignored (missing '=')
thislineisbad

# Example of a line with multiple '=' (will be ignored)
multi=bad=entry

# Example of a version with no digits (will be ignored)
nodigits=latest
